    <script>
        // ฟังก์ชันไฮไลท์สีการ์ดอาหารที่ถูกเลือก
        function selectFood(selectedCard) {
            // ล้างคลาส selected ออกจากการ์ดทุกใบก่อน
            document.querySelectorAll('.food-card').forEach(card => {
                card.classList.remove('selected');
            });
            // เพิ่มคลาส selected ให้ใบที่ถูกคลิก
            selectedCard.classList.add('selected');
        }

        // ระบบส่งข้อมูลเข้า Google Sheets
        document.getElementById('foodForm').addEventListener('submit', function(e) {
            e.preventDefault(); 
            
            const btn = document.getElementById('submitBtn');
            const originalText = btn.innerText;
            btn.innerText = 'กำลังบันทึกออเดอร์...';
            btn.disabled = true;

            const formData = new URLSearchParams(new FormData(this));

            // ⚠️ นำ URL จาก Apps Script แผ่นใหม่ มาใส่ตรงนี้
            const WEB_APP_URL = 'ใส่_URL_ที่ได้จาก_Google_Apps_Script_ไฟล์ใหม่_ตรงนี้';

            fetch(WEB_APP_URL, {
                method: 'POST',
                body: formData,
                mode: 'no-cors'
            })
            .then(() => {
                alert('บันทึกเมนูอาหารเรียบร้อยแล้ว เตรียมตัวอร่อยได้เลย! 🎉');
                this.reset();
                document.querySelectorAll('.food-card').forEach(c => c.classList.remove('selected'));
            })
            .catch(error => {
                alert('เกิดข้อผิดพลาด กรุณาลองใหม่อีกครั้ง');
            })
            .finally(() => {
                btn.innerText = originalText;
                btn.disabled = false;
            });
        });
    </script>
